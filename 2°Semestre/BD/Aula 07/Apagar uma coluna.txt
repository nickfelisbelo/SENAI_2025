/*Apagar uma coluna:*/

ALTER TABLE locacoes(Nome da Tabela) DROP COLUMN preco(Nome da Coluna);



/*Relacionar 2 tabelas:*/

SELECT * FROM locacoes 
INNER JOIN filmes 
ON locacoes.filme_id = filmes.id;



/*Filtar colunas e ordem*/

SELECT locacoes.*, filmes.preco FROM locacoes 
INNER JOIN filmes 
ON locacoes.filme_id = filmes.id
ORDER by locacoes.data_locacao DESC;



/*Mostrar quantidade de locacoes feitas por cliente e ordem crescente (1, 2, 3..., n)*/

SELECT c.*, COUNT(l.cliente_id) as quantidade_locacoes FROM clientes c
INNER JOIN locacoes l
ON l.cliente_id = c.id
GROUP BY c.nome
ORDER by c.id asc;

/*Preco medio por categoria*/

SELECT categoria, AVG(preco) as preco_medio FROM filmes
GROUP by categoria;
